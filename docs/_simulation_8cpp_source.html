77 void
78 Simulation::CreateLinearTopology( void )
79 {
80  ns3::MobilityHelper mobilityHelper;
81  ns3::Ptr<ns3::ListPositionAllocator> positionAlloc = ns3::CreateObject<ns3::ListPositionAllocator>();
82 
83  /* Sink at the center */
84  positionAlloc->Add( ns3::Vector( 0, 0, 0 ) );
85 
86  std::vector<TCoordinate> xList;
87  std::vector<TCoordinate> yList;
88 
89  for ( TNumberOfHops hopIndex = 1; hopIndex <= this->m_numberOfHops; ++hopIndex )
90  {
91  TAngle nodeAngle = 0;
92  TCommunicationRadius hopRadius = this->m_communicationRadius * ( TCommunicationRadius )hopIndex;
93  TCoordinate x = cos( nodeAngle ) * hopRadius;
94  TCoordinate y = sin( nodeAngle ) * hopRadius;
95  xList.push_back( x );
96  yList.push_back( y );
97  positionAlloc->Add( ns3::Vector( x, y, 0.0 ) );
98  }
99 
100  mobilityHelper.SetPositionAllocator( positionAlloc );
101  mobilityHelper.Install( this->m_allNodesContainer );
102 }

104 void
105 Simulation::CreateRadialTopology( void )
106 {
107  ns3::MobilityHelper mobilityHelper;
108  ns3::Ptr<ns3::ListPositionAllocator> positionAlloc = ns3::CreateObject<ns3::ListPositionAllocator>();
109 
110  /* Sink at the center */
111  positionAlloc->Add( ns3::Vector( 0, 0, 0 ) );
112 
113  std::vector<TCoordinate> xList;
114  std::vector<TCoordinate> yList;
115 
116  TNumberOfNodes nodePerHop = std::ceil( 2.0 * ( double )( this->m_numberOfNodes - 1 ) / ( double )( this->m_numberOfHops * ( this->m_numberOfHops + 1 ) ) );
117  std::cout << "Nodes per hop: " << nodePerHop << std::endl;
118  for ( TNumberOfHops hopIndex = 1; hopIndex <= this->m_numberOfHops; ++hopIndex )
119  {
120  TCommunicationRadius hopRadius = this->m_communicationRadius * ( TCommunicationRadius )hopIndex;
121  TNumberOfNodes nodesInThisHop = nodePerHop * hopIndex;
122  for ( TNumberOfNodes nodeIndex = 0; nodeIndex < nodesInThisHop; ++nodeIndex )
123  {
124  TAngle nodeAngle = 2 * M_PI * ( double )nodeIndex / ( double )nodesInThisHop;
125  TCoordinate x = cos( nodeAngle ) * hopRadius;
126  TCoordinate y = sin( nodeAngle ) * hopRadius;
127  xList.push_back( x );
128  yList.push_back( y );
129  positionAlloc->Add( ns3::Vector( x, y, 0.0 ) );
130  }
131  }
132 
133  mobilityHelper.SetPositionAllocator( positionAlloc );
134  mobilityHelper.Install( this->m_allNodesContainer );
135 }

182 void
183 Simulation::InstallWiFi( void )
184 {
185  std::string phyMode( "DsssRate11Mbps" );
186  ns3::Config::SetDefault( "ns3::WifiRemoteStationManager::RtsCtsThreshold", ns3::UintegerValue( 2200 ) );
187  ns3::Config::SetDefault( "ns3::WifiRemoteStationManager::NonUnicastMode", ns3::StringValue( phyMode ) );
188  ns3::YansWifiPhyHelper wifiPhyHelper = ns3::YansWifiPhyHelper::Default();
189  wifiPhyHelper.Set( "RxGain", ns3::DoubleValue( this->m_rxGain ) );
190 
191  ns3::YansWifiChannelHelper wifiChannelHelper;
192  wifiChannelHelper.SetPropagationDelay( "ns3::ConstantSpeedPropagationDelayModel" );
193 
194  if ( this->m_topologyType == this->DowntownTopology )
195  {
196  wifiChannelHelper.AddPropagationLoss( "ns3::HybridBuildingsPropagationLossModel", "Los2NlosThr", ns3::DoubleValue( 1550.0 ) );
197  }
198  else
199  {
200  wifiChannelHelper.AddPropagationLoss( "ns3::RangePropagationLossModel",
201  "MaxRange",
202  ns3::DoubleValue( this->m_communicationRadius * sqrt( 2 ) ) );
203 
204  wifiChannelHelper.AddPropagationLoss( "ns3::FriisPropagationLossModel" );
205  }
206 
207  wifiPhyHelper.SetChannel( wifiChannelHelper.Create() );
208 
209  ns3::WifiHelper wifiHelper;
210  wifiHelper.SetStandard( ns3::WIFI_PHY_STANDARD_80211b );
211 
212  ns3::NqosWifiMacHelper wifiMacHelper = ns3::NqosWifiMacHelper::Default ();
213  wifiHelper.SetRemoteStationManager( "ns3::ConstantRateWifiManager",
214  "DataMode", ns3::StringValue( phyMode ),
215  "ControlMode", ns3::StringValue( phyMode ) );
216 
217  wifiMacHelper.SetType( "ns3::AdhocWifiMac" );
218  this->m_allWirelessDevices = wifiHelper.Install( wifiPhyHelper, wifiMacHelper, this->m_allNodesContainer );
219 }
