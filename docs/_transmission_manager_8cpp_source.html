51 TransmissionManager::CreatePacketFromMeasurements( void )
52 {
53  auto measurementQueue = ns3::Create< MeasurementQueue >();
54  for ( auto i = this->m_applications.begin(); i != this->m_applications.end(); ++i )
55  {
56  ( *i )->GetAllMeasurements( measurementQueue );
57  }
58 
59  unsigned int queueSize = measurementQueue->Size();
60  auto queueBuffer = new unsigned char[ queueSize ];
61  measurementQueue->WriteToBuffer( queueBuffer );
62 
63  TPacketPtr newPacket = ns3::Create< ns3::Packet >( queueBuffer, queueSize );
64  delete[] queueBuffer;
65 
66  return ( newPacket );
67 }
