59 GainFunction::TTransmissionTime
60 FiveGTransmissionManager::GetOptimalTime( void )
61 {
62  if ( this->m_usedCValue != *( this->m_cValue ) )
63  {
64  this->m_usedCValue = *( this->m_cValue );
65  this->m_currentOptimalTime = this->SolveForCValue( *this->m_cValue );
66  }
67  else
68  {
69  GainFunction::TTransmissionTime updatedTime = this->SolveForCValue( *this->m_cValue );
70 
71  if ( std::abs( updatedTime.GetSeconds() - this->m_currentOptimalTime.GetSeconds() ) > this->m_updateThreshold.GetSeconds() )
72  {
73  BaseStation::TResponseVector myResponses, deltaResponses;
74  auto numberOfCValues = BaseStation::GetBaseStation().cVector.size();
75  for ( auto cIndex = 0; cIndex < numberOfCValues; ++cIndex )
76  {
77  myResponses.push_back( this->ProcessCValue( BaseStation::GetBaseStation().cVector[cIndex] ) );
78  deltaResponses.push_back( myResponses[cIndex] - this->m_myResponses[cIndex] );
79  }
80 
81  BaseStation::GetBaseStation().UpdateResponses( deltaResponses );
82  this->m_myResponses = myResponses;
83  }
84  }
85  return ( this->m_currentOptimalTime );
86 }

136 GainFunction::TTransmissionTime
137 FiveGTransmissionManager::SolveForCValue( const TGain cValue ) const
138 {
139  GainFunction::TTransmissionTime xMin = this->m_searchStart;
140  GainFunction::TTransmissionTime xMax = this->m_searchStop;
141  GainFunction::TTransmissionTime xTrial = ( xMin + xMax ) / 2;
142 
143  double yMin = this->GetTotalGainDerivative( xMin ) + cValue;
144  double yMax = this->GetTotalGainDerivative( xMax ) + cValue;
145  double yTrial = this->GetTotalGainDerivative( xTrial ) + cValue;
146 
147  if ( yMin == 0 )
148  {
149  return ( xMin );
150  }
151  else if ( yMax == 0 )
152  {
153  return ( xMax );
154  }
155  else if ( yMin * yMax > 0 )
156  {
157  if ( yMin <= yMax )
158  {
159  return ( xMin );
160  }
161  else
162  {
163  return ( xMax );
164  }
165  }
166 
167  for ( TCount count = 0; count < this->m_searchCount; ++count )
168  {
169  if ( yTrial == 0 )
170  {
171  break;
172  }
173  if ( yTrial * yMin < 0 )
174  {
175  xMax = xTrial;
176  }
177  else
178  {
179  xMin = xTrial;
180  yMin = this->GetTotalGainDerivative( xMin ) + cValue;
181  }
182 
183  xTrial = ( xMin + xMax ) / 2;
184  yTrial = this->GetTotalGainDerivative( xTrial ) + cValue;
185  }
186  return ( xTrial );
187 }
